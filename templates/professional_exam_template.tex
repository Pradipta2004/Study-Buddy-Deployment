\noindent\textbf{Q.1} \hfill \textbf{[1 mark]}
\vspace{0.2cm}

\noindent Which of the following is a Pythagorean triple?

\vspace{0.2cm}
\noindent\begin{enumerate}[label=(\alph*), leftmargin=1.5cm, itemsep=2pt]
    \item 2, 3, 4
    \item 3, 4, 5
    \item 4, 5, 6
    \item 5, 6, 7
\end{enumerate}

\vspace{0.3cm}

\noindent\textbf{Solution:}
\vspace{0.1cm}

\noindent The correct answer is (b) 3, 4, 5 because $3^2 + 4^2 = 9 + 16 = 25 = 5^2$.

\noindent\textbf{Q.3} \hfill \textbf{[2 marks]}
\vspace{0.2cm}

\noindent Solve for $x$: $2x + 5 = 15$

\vspace{0.3cm}

\noindent\textbf{Solution:}
\vspace{0.1cm}

\noindent Given: $2x + 5 = 15$

Step 1: Subtract 5 from both sides
\[2x = 15 - 5 = 10\]

Step 2: Divide both sides by 2
\[x = \frac{10}{2} = 5\]

Therefore, $x = 5$.

\noindent\textbf{Q.5} \hfill \textbf{[5 marks]}
\vspace{0.2cm}

\noindent A rectangular garden has a length of 12 meters and width of 8 meters. 
\begin{enumerate}[label=(\alph*), leftmargin=1.5cm, itemsep=5pt]
    \item Calculate the perimeter of the garden. \hfill [2 marks]
    \item Calculate the area of the garden. \hfill [2 marks]
    \item If a diagonal path is built across the garden, find its length. \hfill [1 mark]
\end{enumerate}

\vspace{0.3cm}

\noindent\textbf{Solution:}
\vspace{0.1cm}

\noindent Given: Length $l = 12$ m, Width $w = 8$ m

\noindent\textbf{(a)} Perimeter of rectangle:
\[P = 2(l + w) = 2(12 + 8) = 2 \times 20 = 40 \text{ meters}\]

\noindent\textbf{(b)} Area of rectangle:
\[A = l \times w = 12 \times 8 = 96 \text{ square meters}\]

\noindent\textbf{(c)} Diagonal length using Pythagorean theorem:
\[d = \sqrt{l^2 + w^2} = \sqrt{12^2 + 8^2} = \sqrt{144 + 64} = \sqrt{208} = 4\sqrt{13} \approx 14.42 \text{ meters}\]
